提示对话框使用例子

  1. $.window.showTip(msg, [options])

    $.window.showTip("欢迎使用artDialog对话框组件！默认2.5秒后将自动关闭！");

  2. $.window.showTip(msg, [options])

    var dialogId = $.window.showTip("欢迎使用artDialog对话框组件！不自动关闭！", {duration:0});

  3. $.window.closeTip(dialogId)

    $.window.closeTip(dialogId);

  4. $.window.showProgressTip(msg)

    var progressDialogId = $.window.showProgressTip("系统正在处理，请稍后...[5秒后进度条将自动关闭]");
window.setTimeout(function(){$.window.closeProgressTip(progressDialogId);}, 5000);

  5. $.window.closeProgressTip(dialogId)

    $.window.closeProgressTip(dialogId);

  6. $.window.showSucceedMsg(msg, [options])

    $.window.showSucceedMsg("操作成功！默认2.5秒后将自动关闭！");

  7. $.window.showErrorMsg(msg, [options])

    $.window.showErrorMsg("系统异常，请与管理员联系！");

  8. $.window.showWarningMsg(msg, [options])

    $.window.showWarningMsg("姓名不能为空！");

  9. $.window.alert(msg, [options])

    $.window.alert("姓名不能为空！");

  10. $.window.confirm(msg, [options])

    $.window.confirm("您确定要删除选中的记录吗？", {
            yesBtnTitle:"删 除",
            noBtnTitle:"关 闭",
            yesHandle:function(){$.window.showTip("您选择了删除按钮！",{duration:1});},
            noHandle:function(){$.window.showTip("您选择了关闭按钮！",{duration:1});}
});

  11. $.window.showMessage(msg, [options])

    $.window.showMessage("自定义消息框的按钮，想定义几个就定义几个！", {
            button:[{
                name: "确 定",
                callback: function(){$.window.showTip("您选择了确定按钮！",{duration:1});},
                focus:true
            },{
                name: "取 消",
                callback: function(){$.window.showTip("您选择了取消按钮！",{duration:1});}
            },{
                name: "不可用",
                callback: function(){},
                disabled: true
            },{
                name: "关 闭",
                callback: function(){$.window.showTip("您选择了关闭按钮！",{duration:1});}
            }],
            title: "自定义消息按钮对话框"
});
